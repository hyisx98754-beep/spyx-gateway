const express = require("express"); const fetch = require("node-fetch"); const cors = require("cors"); const app = express(); app.use(cors()); const nodes = [ "https://spyx-node-a.onrender.com/route", "https://spyx-node-b.onrender.com/route", "https://spyx-node-c.onrender.com/route" ]; let i = 0; app.get("/proxy", async (req, res) => { try { const url = req.query.url; const node = nodes[i]; i = (i + 1) % nodes.length; const r = await fetch(node + "?url=" + encodeURIComponent(url)); const text = await r.text(); res.send(text); } catch (e) { res.send("gateway error"); } }); const PORT = process.env.PORT || 3000; app.listen(PORT, () => console.log("Gateway running"));
