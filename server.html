const express = require("express"); const cors = require("cors"); const app = express(); app.use(cors()); const nodes = [ "https://spyx-node-a.onrender.com/route", "https://spyx-node-b.onrender.com/route", "https://spyx-node-c.onrender.com/route" ]; let i = 0; /* 🔁 ROUND ROBIN ROUTING */ app.get("/proxy", async (req, res) => { try { const url = req.query.url; if (!url) return res.send("no url"); const node = nodes[i]; i = (i + 1) % nodes.length; const response = await fetch(node + "?url=" + encodeURIComponent(url)); const text = await response.text(); res.send(text); } catch (e) { console.error(e); res.send("gateway error"); } }); const PORT = process.env.PORT || 3000; app.listen(PORT, () => { console.log("Gateway running on port " + PORT); });
